# prompt: Rewrite the previous function calc_family_size_adults_vec using vectorization

def calc_family_size_adults_vec(df):
  # Create a boolean mask for rows where Age is 18 or greater
  is_adult = df['Age'] >= 18

  # Calculate family size for all rows
  family_size_all = df['SibSp'] + df['Parch'] + 1

  # Use the boolean mask to select where to apply the calculated family size
  # and where to put 0
  df['FamilySizeAdults_vec'] = family_size_all.where(is_adult, other=0)
  return df

# @title Calculate number of survived based on family size

def survival_by_family_loop(df):
    stats = {}
    for _, row in df.iterrows():
        fsize = row['FamilySize_vec']
        surv  = row['Survived']
        if fsize not in stats:
            stats[fsize] = {'sum': 0}
            stats[fsize]['count'] = 0
        stats[fsize]['sum'] += surv
        stats[fsize]['count'] += 1
    return stats

%timeit survival_by_family_loop(titanic_df)

# @title Average fare paid by females

def avg_fare_loop(df):
  total = count = 0
  for _, r in df.iterrows():
    if r['Sex'] == 'female' and r['Age'] >= 18 and r['FamilySize_vec'] == 1:
      total += r['Fare'];
      count += 1
  return total / count

# prompt: Write a similar function using vectorization

def avg_fare_vec(df):
  # Filter the DataFrame for the specified conditions
  filtered_df = df[(df['Sex'] == 'female') & (df['Age'] >= 18) & (df['FamilySize_vec'] == 1)]
  # Calculate the mean of the 'Fare' column for the filtered DataFrame
  return filtered_df['Fare'].mean()

# prompt: Write functions to calculate the survival probabilty for people with large families (>=4). DO one using loop and another using vectorization

def survival_large_families_loop(df):
    large_family_survived = 0
    large_family_total = 0
    for _, row in df.iterrows():
        if row['FamilySize_vec'] >= 4:
            large_family_total += 1
            if row['Survived'] == 1:
                large_family_survived += 1
    return large_family_survived / large_family_total if large_family_total > 0 else 0

def survival_large_families_vec(df):
    large_families_df = df[df['FamilySize_vec'] >= 4]
    large_family_total = len(large_families_df)
    large_family_survived = large_families_df['Survived'].sum()
    return large_family_survived / large_family_total if large_family_total > 0 else 0

# Example usage:
print("Survival probability for large families (loop):", survival_large_families_loop(titanic_df))
print("Survival probability for large families (vectorization):", survival_large_families_vec(titanic_df))
